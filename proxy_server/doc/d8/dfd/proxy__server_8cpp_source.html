193int ProxyServer::create_target_connection(const std::string& host, int port) {
194 struct addrinfo hints, *result;
195 memset(&hints, 0, sizeof(hints));
196 hints.ai_family = AF_UNSPEC;
197 hints.ai_socktype = SOCK_STREAM;
198
199 std::string port_str = std::to_string(port);
200 if (getaddrinfo(host.c_str(), port_str.c_str(), &hints, &result) != 0) {
201 Logger::get_instance().error("Failed to resolve host: " + host);
202 return -1;
203 }
204
205 int sock = socket(result->ai_family, result->ai_socktype, result->ai_protocol);
206 if (sock < 0) {
207 freeaddrinfo(result);
208 return -1;
209 }
210
211 if (connect(sock, result->ai_addr, result->ai_addrlen) < 0) {
212 Logger::get_instance().error("Failed to connect to target server");
213 close(sock);
214 freeaddrinfo(result);
215 return -1;
216 }
217
218 freeaddrinfo(result);
219 return sock;
220}

230void ProxyServer::tunnel_connection(int client_socket, int target_socket) {
231 fd_set read_fds;
232 char buffer[BUFFER_SIZE];
233
234 while (true) {
235 FD_ZERO(&read_fds);
236 FD_SET(client_socket, &read_fds);
237 FD_SET(target_socket, &read_fds);
238
239 int max_fd = std::max(client_socket, target_socket);
240 if (select(max_fd + 1, &read_fds, nullptr, nullptr, nullptr) < 0) {
241 Logger::get_instance().error("Select error in tunnel");
242 break;
243 }
244
245 // Client to target
246 if (FD_ISSET(client_socket, &read_fds)) {
247 int bytes = recv(client_socket, buffer, sizeof(buffer), 0);
248 if (bytes <= 0) break;
249 if (send(target_socket, buffer, bytes, 0) <= 0) break;
250 }
251
252 // Target to client
253 if (FD_ISSET(target_socket, &read_fds)) {
254 int bytes = recv(target_socket, buffer, sizeof(buffer), 0);
255 if (bytes <= 0) break;
256 if (send(client_socket, buffer, bytes, 0) <= 0) break;
257 }
258 }
259
260 close(target_socket);
261}
